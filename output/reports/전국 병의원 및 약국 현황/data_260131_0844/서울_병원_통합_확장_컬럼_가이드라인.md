## 7. 데이터 활용 가이드

### 7.1 GIS 분석 활용

#### 기본 지도 시각화
```python
import pandas as pd
import folium

# 데이터 로드
df = pd.read_csv('서울_병원_통합_확장_2025.12.csv', encoding='utf-8-sig')

# 좌표 있는 병원만 필터링
df_geo = df[df['좌표(X)'].notna() & df['좌표(Y)'].notna()]

# 지도 생성
m = folium.Map(location=[37.5665, 126.9780], zoom_start=11)

# 마커 추가
for idx, row in df_geo.iterrows():
    folium.Marker(
        [row['좌표(Y)'], row['좌표(X)']],
        popup=f"{row['요양기관명']}<br>{row['종별코드명']}"
    ).add_to(m)

m.save('seoul_hospitals.html')
```
